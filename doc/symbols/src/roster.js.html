1 /**
  2  * Roster Management
  3  * @namespace
  4  * 
  5  * RFC 3921: XMPP IM; Section 7 & 8
  6  * @see http://ietf.org/rfc/rfc3921.txt
  7  */
  8 XC.Roster = {
  9   XMLNS: 'jabber:iq:roster',
 10 
 11   /**
 12    * Request your roster from the server.
 13    * 
 14    * @param {Object}   [callbacks] An Object with 'onError' and 'onSuccess'.
 15    */
 16   request: function (callbacks) {
 17     var iq = XC.XMPP.IQ.extend(),
 18         q = XC.XMPP.Query.extend({xmlns: this.XMLNS});
 19     iq.type('get');
 20     iq.addChild(q);
 21 
 22     this.connection.send(iq.convertToString(), function (packet) {
 23       if (packet.getType() === 'error') {
 24         callbacks.onError(packet);
 25       } else {
 26         var items = packet.getElementsByTagName('item'),
 27             entities = [], idx = items.length,
 28             entity, item, groups, len;
 29         while (idx--) {
 30           item = items[idx];
 31           entity = XC.Entity.extend({
 32             jid: item.getAttribute('jid'),
 33             subscription: item.getAttribute('subscription'),
 34             name: item.getAttribute('name')
 35           });
 36           groups = item.getElemengsByTagName('group');
 37           len = groups ? groups.length : 0;
 38 
 39           if (len) {
 40             entity.groups = [];
 41           }
 42 
 43           while (len--) {
 44             entity.groups.push(groups[len].text);
 45           }
 46           entities.push(entity);
 47         }
 48         this.onSuccess(entities);
 49       }
 50     });
 51   },
 52 
 53   /**
 54    * Add a new entity to your roster.
 55    * (Same as Update + Subscribe.)
 56    * 
 57    * @param {XC.Entity} entity      The entity to add to your roster.
 58    * @param {Object}    [callbacks] An Object with 'onError' and 'onSuccess'.
 59    */
 60   add: function (entity, callbacks) {
 61     this.update(entity, callbacks);
 62     XC.Presence.subscribe(entity, callbacks);
 63   },
 64 
 65   /**
 66    * Update an entity in your roster.
 67    * 
 68    * @param {XC.Entity} entity      The entity to update on your roster.
 69    * @param {Object}    [callbacks] An Object with 'onError' and 'onSuccess'.
 70    */
 71   update: function (entity, callbacks) {
 72     var iq = XC.XMPP.IQ.extend(),
 73         q = XC.XMPP.Query.extend({xmlns: this.XMLNS}),
 74         item = XC.XML.Element.extend({name: 'item'}),
 75         Group = XC.XML.Element.extend({name: 'group'}),
 76         idx = !entity.groups ? 0 : entity.groups.length,
 77         group;
 78     iq.type('set');
 79     item.attr('jid', entity.jid);
 80 
 81     if (entity.name) {
 82       item.attr('name', entity.name);
 83     }
 84 
 85     while (idx--) {
 86       group = Group.extend();
 87       group.text = entity.groups[idx];
 88       item.addChild(group);
 89     }
 90 
 91     iq.addChild(item);
 92     this.connection.send(iq.convertToString(), function (packet) {
 93       if (packet.getType() === 'error') {
 94         callbacks.onError(packet);
 95       } else {
 96         callbacks.onSuccess();
 97       }
 98     });
 99   },
100 
101   /**
102    * Remove an entity from your roster.
103    * 
104    * @param {XC.Entity} entity      The entity to remove from your roster.
105    * @param {Object}    [callbacks] An Object with 'onError' and 'onSuccess'.
106    */
107   remove: function (entity, callbacks) {
108     var iq = XC.XMPP.IQ.extend(),
109         q = XC.XMPP.Query.extend({xmlns: this.XMLNS}),
110         item = XC.XML.Element.extend({name: 'item'});
111 
112     item.attr('jid', entity.jid);
113     item.attr('subscription', 'remove');
114 
115     this.connection.send(iq.convertToString(), function (packet) {
116       if (packet.getType() === 'error') {
117         callbacks.onError(packet);
118       } else {
119         callbacks.onSuccess();
120       }
121     });
122   },
123 
124   /**
125    * Endpoint for a server-side roster push.
126    *
127    * @param {Array} entities An array of entities.
128    */
129   onRosterPush: function (entities) {},
130 
131   /**
132    * Handle incoming out-of-band Roster IQs
133    *
134    * @param {Element} packet The incoming XML stanza.
135    */
136   _handleRoster: function (packet) {
137     var type = packet.getAttribute('type');
138 
139     // Acknowledge a roster push.
140     if (type === 'set') {
141       var iq = XC.XMPP.IQ.extend();
142       iq.type('result');
143       iq.attr('id', packet.getAttribute('id'));
144       this.connection.send(iq.convertToString());
145 
146     // Process the items passed from the roster.
147     } else {
148       var items = packet.getElementsByTagName('item'),
149           entities = [], idx = items.length,
150           entity, item, groups, len;
151 
152       while (idx--) {
153         item = items[idx];
154         entity = XC.Entity.extend({
155           jid: item.getAttribute('jid'),
156           subscription: item.getAttribute('subscription'),
157           name: item.getAttribute('name')
158         });
159         groups = item.getElemengsByTagName('group');
160         len = groups ? groups.length : 0;
161 
162         if (len) {
163           entity.groups = [];
164         }
165 
166         while (len--) {
167           entity.groups.push(groups[len].text);
168         }
169         entities.push(entity);
170       }
171       this.onRosterPush(entities);
172     }
173   }
174 
175 };
