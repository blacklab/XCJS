  2  * Namespace for XML elements
  3  * @namespace

  7 /**
  8  * A simple XML element class.
  9  *
 10  * @example
 11  * var newElement = XC.XML.Element.extend({name: 'foo'})
 12  * newElement.attr('bar', 'bam');
 13  * newElement.addChild(XC.XML.Element.extend({name: 'child'});
 14  *
 15  * @extends XC.Object
 16  * @class
 17  */
 18 XC.XML.Element = XC.Object.extend(/** @lends XC.XML.Element# */{
 19   name: null,
 20   attributes: null,
 21   xmlns: null,
 22   children: null,
 23   text: null,
 24 
 25   /**
 26    * Get or set attributes on the receiver.
 27    *
 28    * @param {String} name The attributes name.
 29    * @param {String} [value] If value is supplied, the attribute will be set.
 30    * @returns {String} the value of the attribute.
 31    */
 32   attr: function (name, value) {
 33     this.attributes = this.attributes || {};
 34     if (value) {
 35       this.attributes[name] = value;
 36     }
 37     return this.attributes[name];
 38   },
 39 
 40   /**
 41    * Add a XML child element to the receiver.
 42    *
 43    * @param {XC.XML.Element} child The XML element to add as a child.
 44    * @returns {XC.XML.Element} The receiver.
 45    */
 46   addChild: function (child) {
 47     this.children = this.children || [];
 48     if (child) {
 49       this.children.push(child);
 50     }
 51     return this;
 52   },
 53 
 54   /**
 55    * Return an XML string representation of this element.
 56    *
 57    * @returns {String} This XML element as XML text.
 58    */
 59   convertToString: function () {
 60     var ret = "";
 61     var attrs = [];
 62 
 63     if (this.xmlns) {
 64       this.attr('xmlns', this.xmlns);
 65     }
 66 
 67     if (this.attributes) {
 68       for (var name in this.attributes) {
 69         var val = this.attributes[name];
 70         if (!val) { 
 71           continue;
 72         }
 73 
 74         attrs.push(name + '="' + val + '"');
 75       }
 76     }
 77 
 78     ret += "<" + this.name;
 79     ret += (attrs.length > 0) ? ' ' + attrs.join(' ') : '';
 80     ret += ">";
 81 
 82     var children = this.children || [];
 83     for (var i = 0, len = children.length; i < len; i++) {
 84       ret += this.children[i].convertToString();
 85     }
 86 
 87     if (this.text) {
 88       ret += this.text;
 89     }
 90 
 91     ret += "</" + this.name + ">";
 92 
 93     return ret;
 94   }
 95 }, /** @lends XC.XML.Element */ {
 96 
 97   /**
 98    * Convenience function for creating a new XC.XML.Element and
 99    * setting attrs and elements in a single function
100    *
101    * @param {Object} [attrs] A hash of attribute names to attribute values.
102    * @param {XC.XML.Element[]} [elements] An array of XC.XML.Element to assign as children.
103    * @returns {XC.XML.Element}
104    */
105   create: function (attrs, elements) {
106     var ret = this.extend();
107 
108     if (attrs) {
109       for (var k in attrs) {
110         if (attrs.hasOwnProperty(k)) {
111           var v = attrs[k];
112           if (!v) {
113             continue;
114           }
115           ret.attr(k, v);
116         }
117       }
118     }
119 
120     elements = (elements && elements.addChild) ? [elements] : elements;
121     if (elements && elements.length) {
122       for (var i = 0, len = elements.length; i < len; i++) {
123         ret.addChild(elements[i]);
124       }
125     }
126 
127     return ret;
128   }

227 /**
228  * XMPP XDataForm element.
229  *
230  * @extends XC.XML.Element
231  * @class
232  */
233 XC.XMPP.XDataForm = XC.XML.Element.extend(/** @lends XC.XMPP.XDataForm# */{
234   name: 'x',
235   xmlns: 'jabber:x:data',
236 
237   type: function (val) {
238     return this.attr('type', val);
239   },
240 
241   /**
242    * A convenience method for adding fields and values to the
243    * XDataForm. Calling this method will add an XDataField and value to
244    * this XDataForm.
245    *
246    * @param {String} name The name of the field, as identified in the 'var' attribute.
247    * @param {String} value The text to insert into the 'value' element.
248    * @param {String} type XDataField type see XEP: 0004.
249    * @returns {XC.XMPP.XDataForm} The receiver.
250    */
251   addField: function (name, value, type) {
252     var f, v;
253     f = XC.XML.Element.extend({name: 'field'});
254     f.attr('var', name);
255 
256     if (value) {
257       v = XC.XML.Element.extend({name: 'value', text: value});
258       f.addChild(v);
259     }
260 
261     if (type) {
262       f.attr('type', type);
263     }
264 
265     return this.addChild(f);
266   }
