1 /**
  2  * Connection object to use for all XC connections. The +initConnection+
  3  * {@link XC.Connection#initConnection} method MUST be called after
  4  * extending this object.
  5  *
  6  * @class
  7  * @extends XC.Object
  8  * @property {XC.Presence} Presence#
  9  * @property {XC.Roster} Roster#
 10  * @property {XC.Chat} Chat#
 11  * @property {XC.Disco} Disco#
 12  */
 13 XC.Connection = XC.Object.extend(/** @lends XC.Connection# */{
 14   /**
 15    * Map of instance names to instance objects. Used during
 16    * initConnection().
 17    *
 18    * @see XC.Connection#initConnection
 19    */
 20   services: {
 21     Presence:   XC.Presence,
 22     Roster:     XC.Roster,
 23     Chat:       XC.Chat,
 24     Disco:      XC.Disco
 25   },
 26 
 27   /**
 28    * Map of jids to event handler functions. Used when message events
 29    * are received from the connection.
 30    *
 31    * @see XC.Connection#registerJIDHandler
 32    * @see XC.Connection#unregisterJIDHandler
 33    */
 34   jidHandlers: {},
 35 
 36   /**
 37    * Initialize the service properties.
 38    *
 39    * @example
 40    * var xc = XC.Connection.extend();
 41    * xc.initConnection();
 42    *
 43    * @return {XC.Connection}
 44    */
 45   initConnection: function () {
 46     if (!this.getJID() || this.getJID() === '') {
 47       throw new XC.Error('Missing JID');
 48     }
 49 
 50     var serviceMap = {};
 51 
 52     for (var s in this.services) {
 53       if (this.services.hasOwnProperty(s)) {
 54         var service = this.services[s];
 55 
 56         this[s] = service.extend({connection: this});
 57       }
 58     }
 59 
 60     // Register for incoming messages.
 61     var that = this;
 62     this.connection.registerHandler('message', function (msg) {
 63       var from = msg.getFrom();
 64       var fn = that.jidHandlers[from];
 65       if (fn) {
 66         fn(msg);
 67       }
 68     });
 69 
 70     return this;
 71   },
 72 
 73   /**
 74    * Sends an XML string to the connection adapter.
 75    *
 76    * @param {String} xml The XML String to send.
 77    * @param {Function} callback Called when a response to this packet is received with the first argument being the received packet.
 78    * @param {Array} [args] An array of arguments to be passed to callback after the packet.
 79    *
 80    * @see XC.ConnectionAdapter#send
 81    */
 82   send: function (xml, callback, args) {
 83     this.connection.send(xml, callback, args || []);
 84   },
 85 
 86   /**
 87    * Returns the JID of this connection.
 88    *
 89    * @example
 90    * xc.getJID();
 91    *
 92    * @returns {String} This connection's JID.
 93    *
 94    * @see XC.ConnectionAdapter#jid
 95    */
 96   getJID: function () {
 97     return this.connection.jid();
 98   }
 99 
100 });
