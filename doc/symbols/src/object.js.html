1 /**
  2  * Object class for XC. All objects inherit from this one.
  3  * @namespace
  4  */
  5 XC.Object = {
  6 
  7   /**
  8    * Iterates over all arguments, adding their own
  9    * properties to the reciever.
 10    * 
 11    * @example
 12    * obj.mixin({param: value});
 13    * 
 14    * @returns {XC.Object} the reciever
 15    * 
 16    * @see XC.Object.extend
 17    */
 18   mixin: function () {
 19     var len = arguments.length;
 20     for (var i = 0; i < len; i++) {
 21       for (var k in arguments[i]) {
 22         if (arguments[i].hasOwnProperty(k)) {
 23           this[k] = arguments[i][k];
 24         }
 25       }
 26     }
 27     return this;
 28   },
 29 
 30   /**
 31    * Creates a new object which extends the current object.
 32    * Any arguments are mixed into the new object as if {@link XC.Object.mixin}
 33    * was called on the new object with remaining args.
 34    * 
 35    * @example
 36    * var obj = XC.Object.extend({param: value});
 37    * 
 38    * @returns {XC.Object} The new object.
 39    * 
 40    * @see XC.Object.mixin
 41    */
 42   extend: function () {
 43     var F = function () {},
 44         rc;
 45     F.prototype = this;
 46     rc = new F();
 47     rc.mixin.apply(rc, arguments);
 48 
 49     if (rc.init && rc.init.constructor === Function) {
 50       rc.init.call(rc);
 51     }
 52 
 53     return rc;
 54   }
 55 
 56 };
