1 /**
  2  * XC Connection Adapter abstract object.
  3  *
  4  * An instance of this object MUST be supplied to the XC.Connection
  5  * instance. This object is to be defined by consumers of the API as
  6  * an adapter to the XMPP connection library that is being used. See
  7  * the example for using the XC.ConnectionAdapter with the JSJaC XMPP
  8  * library.
  9  *
 10  * @example
 11  * var conn = new JSJaCConnection();
 12  * var adapter = XC.ConnectionAdapter.extend({
 13  *   jid: conn.jid,
 14  *
 15  *   registerHandler: function (event, handler) {
 16  *     return conn.registerHandler(event, handler);
 17  *   },
 18  *
 19  *   unregisterHandler: function (event, handler) {
 20  *     return conn.unregisterHandler(event, handler);
 21  *   },
 22  *
 23  *   send: function (xml, cb, args) {
 24  *     return conn._sendRaw(xml, cb, args);
 25  *   }
 26  * });
 27  *
 28  * var tmp = XC.Connection.extend({connection: adapter});
 29  *
 30  * @class
 31  * @extends XC.Object
 32  */
 33 XC.ConnectionAdapter = XC.Object.extend(/** @lends XC.ConnectionAdapter# */{
 34   /** The JID of this connection. */
 35   jid: function () {},
 36 
 37   /**
 38    * Send an XML string to the underlying connection.
 39    *
 40    * @param {String} xml The XML String to send.
 41    * @param {Function} callback Called when a response to this packet is received with the first argument being the received packet.
 42    * @param {Array} args An array of arguments to be passed to callback after the packet.
 43    *
 44    * @see XC.Connection#send
 45    */
 46   send: function (xml, callback, args) {},
 47 
 48   /**
 49    * Registers an event handler.
 50    *
 51    * @param {String} event The type of stanza for which to listen (i.e., `message', `iq', `presence.')
 52    * @param {Function} handler The stanza is passed to this function when it is received.
 53    *
 54    * @see XC.ConnectionAdapter#unregisterHandler
 55    * @see XC.Connection#registerJIDHandler
 56    */
 57   registerHandler: function (event, handler) {},
 58 
 59   /**
 60    * Unregisters an event handler.
 61    *
 62    * @param {String} event The type of stanza we were listening to (i.e., `message', `iq', `presence.')
 63    *
 64    * @see XC.ConnectionAdapter#registerHandler
 65    * @see XC.Connection#unregisterJIDHandler
 66    */
 67   unregisterHandler: function (event) {}
 68 });
