1 /**
  2  * Service Discovery provides the ability to discover information about entities.
  3  * @namespace
  4  * 
  5  * XEP-0030: Service Discovery
  6  * @see http://xmpp.org/extensions/xep-0030.html
  7  * 
  8  * @requires rootItem  A list of features and items associated with your entity.
  9  */
 10 XC.Disco = {
 11 
 12   XMLNS: 'http://jabber.org/protocol/disco',
 13 
 14   /**
 15    * @param {XC.Entity} entity
 16    * @param {Object}    [callbacks]
 17    */
 18   info: function (entity, callbacks) {},
 19 
 20   /**
 21    * @param {XC.Entity} entity
 22    * @param {Object}    [callbacks]
 23    */
 24   items: function (entity, callbacks) {},
 25 
 26   /**
 27    * The root item of the Disco Item tree.
 28    * @type {XC.DiscoItem}
 29    */
 30   rootItem: null
 31 };
 32 
 33 /**
 34  * Represents an item (node) in Service Discovery.
 35  * 
 36  * @extends XC.Object
 37  * @class
 38  * @see XC.Disco
 39  */
 40 XC.DiscoItem = XC.Object.extend(/** @lends XC.DiscoItem */{
 41 
 42   /**
 43    * The name of the node.
 44    * @type {String}
 45    */
 46   name: '',
 47 
 48   /**
 49    * Namespaces of supported features.
 50    * @type {String}
 51    */
 52   features: [],
 53 
 54   /**
 55    * Associative array of all items in this item.
 56    * @type {XC.DiscoItem}
 57    */
 58   items: {},
 59 
 60   /**
 61    * Identities that this item represents.
 62    * @type {Object}
 63    */
 64   identities: [],
 65 
 66   /**
 67    * Add a feature to this item.
 68    * @param {String} xmlns The namespace of the feature to add.
 69    */
 70   addFeature: function (xmlns) {},
 71 
 72   /**
 73    * Remove a pre-existing feature from this item.
 74    * @param {String} xmlns The namespace of the feature to remove.
 75    */
 76   removeFeature: function (xmlns) {},
 77 
 78   /**
 79    * Add a child item to this item.
 80    * @param {XC.DiscoItem} discoItem The item to add.
 81    */
 82   addItem: function (discoItem) {},
 83 
 84   /**
 85    * Remove a pre-existing item from this item.
 86    * @param {XC.DiscoItem} discoItem The item to remove.
 87    */
 88   removeItem: function (discoItem) {},
 89 
 90   /**
 91    * Add an identity to this item.
 92    * {
 93    *   category: {String},
 94    *   name: {String},
 95    *   type: {String}
 96    * }
 97    * @param {Object} identity The identity to add.
 98    */
 99   addIdentity: function (identity) {},
100 
101   /**
102    * Remove a pre-existing identity from this item.
103    * @param {Object} identity The identity to remove.
104    */
105   removeIdentity: function (identity) {},
106 
107   /**
108    * Disco items request on this item.
109    * @param {XC.Entity} entity
110    */
111   getItems: function (entity) {},
112 
113   /**
114    * Disco info request on this item.
115    * @param {XC.Entity} entity
116    */
117   getInfo: function (entity) {}
118 
119 });
