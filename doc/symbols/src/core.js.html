  7 var XC = {

 11   },
 12 
 13   log: function () {
 14     var message = Array.prototype.join.apply(arguments, [' ']);
 15     return window.console && window.console.log && window.console.log.apply(window.console, [message]);
 16   },
 17 
 18   warn: function () {
 19     var message = Array.prototype.join.apply(arguments, [' ']);
 20     return window.console && window.console.warn && window.console.warn.apply(window.console, [message]);
 21   },
 22 
 23   error: function () {
 24     var message = Array.prototype.join.apply(arguments, [' ']);
 25     return window.console && window.console.error && window.console.error.apply(window.console, [message]);
 26   },
 27 
 28   group: function () {
 29     if (window.console && window.console.group) {
 30       window.console.group.apply(window.console, arguments);
 31     } else {
 32       XC.log.apply(XC, arguments);
 33     }
 34   },
 35 
 36   groupEnd: function () {
 37     if (window.console && window.console.groupEnd) {
 38       window.console.groupEnd();
 39     }
 40   }
