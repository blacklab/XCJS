1 /**
  2  * Presence
  3  * @namespace
  4  * 
  5  * RFC 3921: XMPP IM; Section 5 & 6
  6  * @see http://www.ietf.org/rfc/rfc3921.txt
  7  */
  8 XC.Presence = {
  9 
 10   SHOW: {
 11     AWAY: 'away',  // The entity or resource is temporarily away.
 12     CHAT: 'chat',  // The entity or resource is actively interested in chatting.
 13     DND:  'dnd',   // The entity or resource is is busy (dnd = "Do Not Disturb").
 14     XA:   'xa'     // The entity or resource is away for an extended period 
 15                    // (xa = "eXtended Away").
 16   },
 17 
 18   // In band
 19 
 20   /**
 21    * Send presence to all subscribed entities / resources
 22    * or send direced presence to a specific entity.
 23    * 
 24    * @param {XC.Entity} [entity]    The entity to direct presence to.
 25    * @param {Object}    [callbacks] An Object with 'onError'.
 26    */
 27   send: function (entity, callbacks) {
 28     var p = XC.XMPP.Presence.extend(),
 29         presence = this.presence;
 30 
 31     // Send directed presence.
 32     if (entity) {
 33       p.to(entity.jid);
 34     }
 35 
 36     if (this.status) {
 37       var status = XC.XML.Element.extend({
 38         name: 'status'
 39       });
 40       status.text = presence.status.toString();
 41       p.addChild(status);
 42     }
 43 
 44     if (presence.show !== XC.Presence.SHOW.AVAILABLE) {
 45       var show = XC.XML.Element.extend({
 46         name: 'show'
 47       });
 48 
 49       // Show must be one of the pre-defined constants
 50       if (XC.IM.Presence.SHOW[presence.show.toUpperCase()]) {
 51         show.text = presence.show;
 52         p.addChild(show);
 53       }
 54     }
 55 
 56     if (presence.priority) {
 57       var priority = XC.XML.Element.extend({
 58         name: 'priority'
 59       }), iPriority = parseInt(presence.priority, 10);
 60 
 61       // The priority MUST be an integer between -128 and +127
 62       if (iPriority > -128 && iPriority < 127) {
 63         priority.text = presence.priority;
 64         p.addChild(priority);
 65       }
 66     }
 67 
 68     this.connection.send(p.convertToString(), function (packet) {
 69       if (packet.getType() === 'error') {
 70         callbacks.onError(packet);
 71       }
 72     });
 73   },
 74 
 75   /**
 76    * Send 'unavailable' presence.
 77    */
 78   unavailable: function () {
 79     var p = XC.XMPP.Presence.extend();
 80     p.type('unavailable');
 81     this.connection.send(p.convertToString());
 82   },
 83   
 84   /**
 85    * Request a subscription to an entity's presence.
 86    * 
 87    * @param {XC.Entity} entity      The entity to request a presence subscription from.
 88    * @param {Object}    [callbacks] An Object with 'onError'.
 89    */
 90   subscribe: function (entity, callbacks) {
 91     var p = XC.XMPP.Presence.extend();
 92     p.attr('type', 'subscribe');
 93     p.to(entity.jid);
 94 
 95     this.connection.send(p.convertToString(), function (packet) {
 96       if (packet.getType() === 'error') {
 97         callbacks.onError();
 98       }
 99     });
100   },
101 
102   /**
103    * Unsubscribe from an entity's presence.
104    * 
105    * @param {XC.Entity} entity      The entity to unsubscribe from it's presence.
106    * @param {Object}    [callbacks] An Object with 'onError'.
107    */
108   unsubscribe: function (entity, callbacks) {
109     var p = XC.XMPP.Presence.extend();
110     p.attr('type', 'unsubscribe');
111     p.to(entity.jid);
112 
113     this.connection.send(p.convertToString(), function (packet) {
114       if (packet.getType() === 'error') {
115         callbacks.onError();
116       }
117     });
118   },
119 
120   // Out of band
121 
122   /**
123    * Approve a pending subscription request from an entity.
124    * 
125    * @param {XC.Entity} entity      The entity requesting a presence subscription.
126    */
127   approveSubscription: function (entity) {
128     var p = XC.XMPP.Presence.extend();
129     p.attr('type', 'subscribed');
130     p.to(entity.jid);
131 
132     this.connection.send(p.convertToString());
133   },
134 
135   /**
136    * Deny a pending subscription request from an entity.
137    * 
138    * @param {XC.Entity} entity      The entity requesting a presence subscription.
139    */
140   denySubscription: function (entity) {
141     var p = XC.XMPP.Presence.extend();
142     p.attr('type', 'unsubscribed');
143     p.to(entity.jid);
144 
145     this.connection.send(p.convertToString());
146   },
147 
148   /**
149    * Endpoint for requests to subscribe to your presence.
150    * 
151    * @param {XC.Entity} entity      The entity requesting a presence subscription.
152    */
153   onSubscribe: function (entity) {},
154 
155   /**
156    * Endpoint notifying that you are subscribed to the entity's presence.
157    * 
158    * @param {XC.Entity} entity      The entity whose presence you are subscribed to.
159    */
160   onSubscribed: function (entity) {},
161 
162   /**
163    * Endpoint notifying that you are unsubscribed from the entity's presence.
164    * 
165    * @param {XC.Entity} entity      The entity whose presence you are unsubscribed from.
166    */
167   onUnsubscribed: function (entity) {},
168 
169   /**
170    * Handle out-of-band presence stanzas
171    *
172    * @param {Element} packet The incoming XML stanza.
173    */
174   _handlePresence: function (packet) {
175     var jid = packet.getAttribute('from'),
176         type = packet.getAttribute('type'),
177         entity = XC.Entity.extend({jid: jid});
178 
179     switch (type) {
180     case 'error':
181       break;
182     case 'probe':
183       break;
184     case 'subscribe':
185       this.onSubscribe(entity);
186       break;
187     case 'subscribed':
188       this.onSubscribed(entity);
189       break;
190     case 'unsubscribe':
191       break;
192     case 'unsubscribed':
193       this.onUnsubscribed(entity);
194       break;
195     }
196   }
197   
198 };
